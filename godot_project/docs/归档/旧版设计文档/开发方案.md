## 二、技术架构（专家建议版）

### 2.1 架构原则

**核心思想**：数据驱动 + 组件化 + 事件解耦

- **Component**：挂在实体上，保存自有状态，不写业务逻辑
- **System/Manager**：处理跨实体规则和流程
- **Calculator**：纯函数计算，不碰节点
- **Resource**：静态配置，只读不写

## 四、ATB与战斗时钟设计（v4.0子弹时间版）

### 4.1 BattleClock（战斗时钟）- 核心重构

解决"选牌时ATB是否暂停"的核心矛盾，**采用子弹时间而非完全暂停**：

```gdscript
# systems/combat/battle_clock.gd
class_name BattleClock
extends Node

enum State { RUNNING, BULLET_TIME, PAUSED, FROZEN }

var state: State = State.RUNNING
var time_scale: float = 1.0
var bullet_time_scale: float = 0.1  # 子弹时间0.1x

var time_sand: int = 3  # 时砂资源，每局初始3点
var time_sand_max: int = 3

signal state_changed(new_state: State)
signal time_sand_changed(current: int, max: int)

func enter_bullet_time():
    """进入子弹时间（0.1x慢速）"""
    state = State.BULLET_TIME
    state_changed.emit(state)

func use_time_sand_pause():
    """消耗时砂进入真正暂停（3秒）"""
    if time_sand > 0:
        time_sand -= 1
        time_sand_changed.emit(time_sand, time_sand_max)
        state = State.PAUSED
        state_changed.emit(state)
        await get_tree().create_timer(3.0).timeout
        resume()
    else:
        # 时砂耗尽，只能0.1x子弹时间
        enter_bullet_time()

func resume():
    """恢复正常速度"""
    state = State.RUNNING
    time_scale = 1.0
    state_changed.emit(state)

func freeze():
    """ATB冻结（溢出后）"""
    state = State.FROZEN
    state_changed.emit(state)

func get_effective_delta(delta: float) -> float:
    """获取有效delta，根据状态计算"""
    match state:
        State.RUNNING:
            return delta * time_scale
        State.BULLET_TIME:
            return delta * bullet_time_scale  # 0.1x慢速
        State.PAUSED, State.FROZEN:
            return 0.0
    return 0.0

func add_time_sand(amount: int):
    """增加时砂（击杀敌人掉落）"""
    time_sand = min(time_sand + amount, time_sand_max)
    time_sand_changed.emit(time_sand, time_sand_max)
```

### 4.2 ATBComponent（组件）- 支持速度软上限

```gdscript
# entities/components/atb_component.gd
class_name ATBComponent
extends Node

@export var base_speed: float = 100.0
var bonus_speed: float = 0.0  # 来自装备/词缀的额外速度
var kinetic_energy: float = 0.0  # 动能（速度超200时转化）

var atb_value: float = 0.0
var max_atb: float = 1000.0

var _last_emitted: float = -1.0
var _threshold: float = 5.0  # 变化阈值

signal atb_full(entity)
signal atb_changed(value, max)

func get_total_speed() -> float:
    """计算总速度，考虑软上限"""
    var total = base_speed + bonus_speed
    if total > 200:
        # 超出的部分转化为动能
        kinetic_energy += (total - 200) * 0.01
        return 200.0
    return total

func _process(delta: float):
    var effective_delta = BattleClock.get_effective_delta(delta)
    if effective_delta <= 0.0:
        return

    var speed = get_total_speed()
    atb_value = min(atb_value + speed * effective_delta * 10, max_atb)

    if abs(atb_value - _last_emitted) > _threshold or atb_value >= max_atb:
        atb_changed.emit(atb_value, max_atb)
        _last_emitted = atb_value

    if atb_value >= max_atb:
        atb_full.emit(get_parent())

func reset():
    atb_value = 0.0
    _last_emitted = -1.0

func apply_kinetic_energy() -> float:
    """应用动能，返回伤害加成百分比"""
    if kinetic_energy > 0:
        var bonus = kinetic_energy
        kinetic_energy = 0.0
        return bonus
    return 0.0
```

### 4.3 行动队列（新增）

```gdscript
# systems/combat/action_queue.gd
class_name ActionQueue
extends Node

const MAX_QUEUE_SIZE = 3

var queued_actions: Array[SkillInstance] = []

signal queue_changed(queue_size: int)
signal actions_executed(actions: Array)

func enqueue(skill: SkillInstance) -> bool:
    """将技能加入行动队列"""
    if queued_actions.size() < MAX_QUEUE_SIZE:
        queued_actions.append(skill)
        queue_changed.emit(queued_actions.size())
        return true
    return false

func clear():
    queued_actions.clear()
    queue_changed.emit(0)

func execute_all() -> Array:
    """执行队列中的所有动作"""
    var executed = queued_actions.duplicate()
    queued_actions.clear()
    queue_changed.emit(0)
    actions_executed.emit(executed)
    return executed
```

### 4.4 战斗状态机（更新版）

```gdscript
# systems/combat/battle_state.gd
class_name BattleState
extends Node

enum State {
    INIT,              # 初始化
    RUNNING,           # ATB运行中（正常速度）
    BULLET_TIME,       # 子弹时间（0.1x慢速）
    PLAYER_INPUT,      # 玩家输入阶段
    EXECUTING_QUEUE,   # 执行行动队列
    ENEMY_TURN,        # 敌人行动
    RESOLVING,         # 结算中
    FROZEN,            # ATB冻结（溢出）
    ENDED              # 战斗结束
}

var current_state: State = State.INIT
signal state_changed(from_state: State, to_state: State)

func transition_to(new_state: State):
    state_changed.emit(current_state, new_state)
    current_state = new_state

    # 状态转换时通知BattleClock
    match new_state:
        State.BULLET_TIME:
            BattleClock.enter_bullet_time()
        State.PLAYER_INPUT:
            BattleClock.enter_bullet_time()
        State.RUNNING:
            BattleClock.resume()
        State.FROZEN:
            BattleClock.freeze()
```

```gdscript
# core/enums.gd
extends Node

# 元素类型
enum Element { FIRE, ICE, THUNDER, WIND, EARTH, PHYSICAL }

# 元素反应类型（15种二阶反应）
enum ElementReaction {
    BURNING_WIND,      # 火+风
    EVAPORATION,       # 火+冰
    FLAME_LIGHTNING,  # 火+雷
    MAGMA,            # 火+土
    SUPERCONDUCT,     # 冰+雷
    COLD_CURRENT,     # 冰+风
    FROZEN_EARTH,     # 冰+土
    IONIZATION,       # 雷+风
    MAGNETIZATION,    # 雷+土
    SANDSTORM,        # 风+土
    BLAZING,          # 火+火
    ABSOLUTE_ZERO,    # 冰+冰
    THUNDER_ACCUM,    # 雷+雷
    GALE_STRENGTH,    # 风+风
    EARTHEN_TREMOR,   # 土+土
    SYNERGY           # 元素共鸣（三阶）
}

# 装备槽位
enum EquipmentSlot { WEAPON, ARMOR, ACCESSORY_1, ACCESSORY_2, GEM_1, GEM_2, GEM_3 }

# 装备稀有度
enum Rarity { WHITE, GREEN, BLUE, PURPLE, ORANGE, RED }

# 装备路线
enum Route { CULTIVATION, TECHNOLOGY, NEUTRAL }

# 词缀类型
enum AffixType { DAMAGE, ATB, ENERGY, ELEMENT_REACTION, SPECIAL, SET_BONUS }

# 技能类型
enum SkillType { ATTACK, DEFENSE, SUPPORT, ULTIMATE }

# 效果目标
enum EffectTarget { SINGLE, ALL, SELF }

# 获取元素克制关系
func get_element_advantage(element: Element) -> Element:
    match element:
        FIRE: return ICE
        ICE: return FIRE
        THUNDER: return WIND
        WIND: return EARTH
        EARTH: return THUNDER
    return PHYSICAL

# 获取反应类型（用于ElementSystem）
func get_reaction_type(elem1: Element, elem2: Element) -> ElementReaction:
    if elem1 == elem2:
        match elem1:
            FIRE: return ElementReaction.BLAZING
            ICE: return ElementReaction.ABSOLUTE_ZERO
            THUNDER: return ElementReaction.THUNDER_ACCUM
            WIND: return ElementReaction.GALE_STRENGTH
            EARTH: return ElementReaction.EARTHEN_TREMOR
    else:
        var combo = [mini(elem1, elem2), maxi(elem1, elem2)]
        if combo == [FIRE, ICE]: return ElementReaction.EVAPORATION
        if combo == [FIRE, THUNDER]: return ElementReaction.FLAME_LIGHTNING
        if combo == [FIRE, WIND]: return ElementReaction.BURNING_WIND
        if combo == [FIRE, EARTH]: return ElementReaction.MAGMA
        if combo == [ICE, THUNDER]: return ElementReaction.SUPERCONDUCT
        if combo == [ICE, WIND]: return ElementReaction.COLD_CURRENT
        if combo == [ICE, EARTH]: return ElementReaction.FROZEN_EARTH
        if combo == [THUNDER, WIND]: return ElementReaction.IONIZATION
        if combo == [THUNDER, EARTH]: return ElementReaction.MAGNETIZATION
        if combo == [WIND, EARTH]: return ElementReaction.SANDSTORM
    return ElementReaction.SYNERGY
```

```gdscript
# data/equipment/affix_definition.gd
class_name AffixDefinition
extends Resource

enum AffixType {
    DAMAGE,           # 伤害类
    ATB,              # ATB类
    ENERGY,           # 能量类
    ELEMENT_REACTION,  # 元素反应类
    SPECIAL,           # 特殊类
    SET_BONUS         # 套装词缀
}

enum AffixTier { COMMON, RARE, LEGENDARY }  # 词缀层级

@export var id: StringName
@export var name: String
@export var description: String
@export var type: AffixType
@export var tier: AffixTier

@export var value: float  # 效果数值
@export var value2: float  # 第二个数值（如范围、概率）

# 效果参数
@export var effect_modifier: String  # 影响的属性
@export var condition: String  # 触发条件

# 分层设计（核心/扩展/稀有）
@export var is_core: bool = false  # 是否为核心词缀
@export var exclusive_with: Array[StringName]  # 互斥词缀ID列表

func get_display_value() -> String:
    """返回格式化的显示数值"""
    match type:
        AffixType.DAMAGE:
            return "+%.0f%% %s伤害" % [value * 100, effect_modifier]
        AffixType.ATB:
            if effect_modifier == "speed":
                return "+%.0f%% ATB速度" % [value * 100]
            elif effect_modifier == "drain":
                return "ATB倒退%.0f%%" % [value * 100]
        AffixType.ENERGY:
            return "能量上限+%.0f" % [value]
    return str(value)
```

```gdscript
class_name ElementComponent
extends Node

var attached_elements: Dictionary = {}  # {element_type: stack_count}
var element_timers: Dictionary = {}  # 元素持续时间

signal element_attached(element: int, stack: int)
signal element_removed(element: int)
signal reaction_triggered(reaction_type: int, elements: Array)

const REACTION_COOLDOWN = 5.0  # 元素反应冷却（秒）
var reaction_cooldowns: Dictionary = {}

func apply_element(element: Enums.Element, stacks: int = 1, duration: float = 0.0):
    """附着元素"""
    if attached_elements.has(element):
        attached_elements[element] += stacks
    else:
        attached_elements[element] = stacks

    if duration > 0:
        element_timers[element] = duration

    element_attached.emit(element, attached_elements[element])

func check_reaction(other_element: Enums.Element) -> int:
    """检查是否能触发反应"""
    if not attached_elements.has(other_element):
        return -1

    # 检查冷却
    var reaction_key = "%d_%d" % [mini(element, other_element), maxi(element, other_element)]
    if reaction_cooldowns.has(reaction_key):
        return -1

    var reaction_type = ElementSystem.get_reaction_type(element, other_element)
    if reaction_type >= 0:
        reaction_cooldowns[reaction_key] = REACTION_COOLDOWN
        reaction_triggered.emit(reaction_type, [element, other_element])

    return reaction_type

func get_element_stacks(element: Enums.Element) -> int:
    return attached_elements.get(element, 0)

func clear_all():
    attached_elements.clear()
    element_timers.clear()
```

### 7.3 技能系统（SkillSystem - 替代DeckSystem）

```gdscript
# systems/skill/skill_system.gd
class_name SkillSystem
extends Node

# 当前装备提供的技能
var available_skills: Array[SkillInstance] = []

# 手牌（技能）
var hand: Array[SkillInstance] = []
const MAX_HAND_SIZE = 8

# 行动队列
var action_queue: ActionQueue

signal hand_changed(hand: Array)
signal skill_executed(skill: SkillInstance)

func setup_from_equipment(equipment: Array[EquipmentInstance]):
    """根据当前装备刷新可用技能"""
    available_skills.clear()
    for equip in equipment:
        if equip and equip.definition:
            for skill_id in equip.definition.skill_ids:
                var skill_def = DataManager.get_skill(skill_id)
                if skill_def:
                    var inst = SkillInstance.new()
                    inst.setup(skill_def, equip.definition.id)
                    available_skills.append(inst)

    # 刷新手牌
    _refresh_hand()

func _refresh_hand():
    """刷新手牌，补充到最大手牌数"""
    while hand.size() < MAX_HAND_SIZE and available_skills.size() > 0:
        var skill = available_skills.pop_front()
        hand.append(skill)
    hand_changed.emit(hand)

func play_skill(skill: SkillInstance, target) -> bool:
    """使用技能"""
    if not skill.is_ready():
        return false

    var cost = skill.get_actual_cost()
    if not EnergySystem.try_consume(cost):
        return false

    skill.on_skill_used()
    _execute_skill(skill, target)

    # 检查连携
    _check_chain(skill)

    # 重新补充手牌
    _refresh_hand()
    return true

func _execute_skill(skill: SkillInstance, target):
    """执行技能效果"""
    for effect in skill.definition.effects:
        effect.apply(skill, target, _build_context(skill))

    skill_executed.emit(skill)

func _check_chain(skill: SkillInstance):
    """检查技能连携"""
    if skill.definition.chain_skill_id.is_empty():
        return

    var chain_def = DataManager.get_skill(skill.definition.chain_skill_id)
    if chain_def:
        # 触发连携特效
        EventBus.skill.skill_chain_triggered.emit(skill.definition, chain_def)
        # 连携技能消耗动能
        if EnergySystem.try_consume_kinetic(0.1):  # 消耗10%动能
            # 执行连携
            available_skills.append(SkillInstance.new().setup(chain_def))

func _build_context(skill: SkillInstance) -> Dictionary:
    return {
        "kinetic_energy": ATBComponent.kinetic_energy if ATBComponent else 0.0,
        "atb_timing": _calculate_atb_timing(),
        "source_equipment": skill.from_equipment
    }

func _calculate_atb_timing() -> float:
    """计算ATB时机加成"""
    var atb = ATBComponent.atb_value / ATBComponent.max_atb
    if atb >= 0.9:
        return 1.15  # 完美时机
    elif atb >= 0.7:
        return 1.0   # 正常
    else:
        return 0.8   # 仓促
```

```gdscript
# systems/skill/energy_system.gd
class_name EnergySystem
extends Node

var current_energy: int = 3
var max_energy: int = 5
var kinetic_energy: float = 0.0

signal energy_changed(current: int, max: int)
signal kinetic_changed(amount: float)

func _ready():
    EventBus.combat.atb_full.connect(_on_player_atb_full)

func _on_player_atb_full(entity):
    """ATB满时恢复能量"""
    if entity is Player:
        _restore_energy(3)
        kinetic_energy += (ATBComponent.get_total_speed() - 100) * 0.005
        kinetic_changed.emit(kinetic_energy)

func _restore_energy(amount: int):
    current_energy = min(current_energy + amount, max_energy)
    energy_changed.emit(current_energy, max_energy)

func try_consume(amount: int) -> bool:
    if current_energy >= amount:
        current_energy -= amount
        energy_changed.emit(current_energy, max_energy)
        return true
    return false

func try_consume_kinetic(amount: float) -> bool:
    """消耗动能"""
    if kinetic_energy >= amount:
        kinetic_energy -= amount
        kinetic_changed.emit(kinetic_energy)
        return true
    return false
```
